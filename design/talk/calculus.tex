\begin{frame}
  \frametitle{Simple example}
  Let $\A_n = \frac1{\sqrt n}\I_n$ be a sequence of (deterministic) matrices. Then, we have:
  \begin{align*}
    \A_n \asymp \zero_n.
  \end{align*}
  Now, let $\C_n = \frac1q \sum_{k=1}^q\A_{n,k}$, where $\A_{n,k}$ are
  copies of $\A_n$.\\
  Trivially, it holds that:
  \begin{align*}
    \C_n^\top\C_n \asymp \frac1{q}\A_{n,1}^\top\A_{n,1} \asymp \zero.
  \end{align*}
  Moreover, we have:
  \begin{align*}
    \tr(\C_n^\top\C_n) = 1.
  \end{align*}
\end{frame}
